```python
class LoRALinear(nn.Module):
    # members: self.base (frozen nn.Linear), self.scale = alpha / r (SINGLE source of truth),
    # self.lora_A (r, in_features), self.lora_B (out_features, r), self.dropout,
    # self.enabled = True, self.merged = False   (plain Python bools — never in state_dict)
    def forward(self, x):
        y = self.base(x)
        if self.enabled and not self.merged:   # D-05 gate — already shipped
            y = y + self.scale * (self.dropout(x) @ self.lora_A.T @ self.lora_B.T)
        return y
```
